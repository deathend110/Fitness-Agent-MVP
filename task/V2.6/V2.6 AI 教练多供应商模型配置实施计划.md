# V2.6 AI 教练多供应商模型配置 Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** 为 RepMind AI 教练接入独立 JSON 模型配置、OpenAI-compatible 与 Gemini-native 双协议 Provider、完整 tool calling 闭环，以及保存后无需重启的热刷新能力。

**Architecture:** 后端把当前 `DeepSeekClient + /api/models + chat/tool_calling/background_worker` 升级为“启动配置层 + Provider 配置服务层 + Provider 适配层 + 协议无关 Tool Loop Orchestrator”。前端新增模型设置弹窗，改为消费后端返回的可用模型集合与模型级 thinking 能力，并让聊天、草稿和后台任务统一使用 `modelRef`。

**Tech Stack:** FastAPI, Pydantic Settings, SQLAlchemy AsyncSession, httpx, React, Tailwind CSS, node --test, pytest, Playwright

# backend/model_config/types.py
from pydantic import BaseModel, Field


class SelectedModelConfig(BaseModel):
    remote_id: str = Field(alias="remoteId")
    label: str
    enabled: bool = True


class ProviderConfig(BaseModel):
    id: str
    type: str
    label: str
    enabled: bool = True
    api_key: str = Field(alias="apiKey")
    base_url: str = Field(alias="baseUrl")
    selected_models: list[SelectedModelConfig] = Field(default_factory=list, alias="selectedModels")


class ProviderConfigDocument(BaseModel):
    version: int = 1
    default_model_ref: str = Field(alias="defaultModelRef")
    providers: list[ProviderConfig]

# backend/model_config/service.py
import json
from pathlib import Path
from typing import Any

from backend.model_config.types import ProviderConfigDocument


class ModelProviderConfigService:
    def __init__(self, *, config_path: Path, legacy_settings: dict[str, Any] | None = None) -> None:
        self.config_path = config_path
        self.legacy_settings = legacy_settings or {}

    def load(self) -> ProviderConfigDocument:
        if self.config_path.exists():
            return ProviderConfigDocument.model_validate_json(self.config_path.read_text(encoding="utf-8"))
        document = self._build_legacy_document()
        self._write_document(document)
        return document

    def save(self, payload: dict[str, Any]) -> dict[str, Any]:
        document = ProviderConfigDocument.model_validate(payload)
        self._write_document(document)
        masked = document.model_dump(by_alias=True)
        for provider in masked["providers"]:
            api_key = provider.pop("apiKey", "")
            provider["hasApiKey"] = bool(api_key)
            provider["apiKeyPreview"] = f"{api_key[:4]}...{api_key[-4:]}" if api_key else ""
        return masked

    def _build_legacy_document(self) -> ProviderConfigDocument:
        allowlist = self.legacy_settings.get("model_allowlist") or ["deepseek-v4-flash"]
        default_model = self.legacy_settings.get("default_model") or allowlist[0]
        payload = {
            "version": 1,
            "defaultModelRef": f"provider_deepseek_default::{default_model}",
            "providers": [
                {
                    "id": "provider_deepseek_default",
                    "type": "openai_compatible",
                    "label": "DeepSeek 默认配置",
                    "enabled": True,
                    "apiKey": self.legacy_settings.get("deepseek_api_key", ""),
                    "baseUrl": self.legacy_settings.get("deepseek_base_url", "https://api.deepseek.com"),
                    "selectedModels": [
                        {"remoteId": model_id, "label": model_id, "enabled": True} for model_id in allowlist
                    ],
                }
            ],
        }
        return ProviderConfigDocument.model_validate(payload)

    def _write_document(self, document: ProviderConfigDocument) -> None:
        self.config_path.parent.mkdir(parents=True, exist_ok=True)
        temp_path = self.config_path.with_suffix(".tmp")
        temp_path.write_text(json.dumps(document.model_dump(by_alias=True), ensure_ascii=False, indent=2), encoding="utf-8")
        temp_path.replace(self.config_path)

# backend/model_config/runtime.py
from dataclasses import dataclass
from pathlib import Path

from backend.model_config.service import ModelProviderConfigService


@dataclass(frozen=True)
class ResolvedModelRef:
    provider: object
    remote_model_id: str


class ProviderRuntimeCache:
    def __init__(self, *, config_path: Path) -> None:
        self.config_path = config_path
        self._service = ModelProviderConfigService(config_path=config_path)
        self.refresh()

    @classmethod
    def from_path(cls, config_path: Path) -> "ProviderRuntimeCache":
        return cls(config_path=config_path)

    def refresh(self) -> None:
        self.document = self._service.load()
        self.default_model_ref = self.document.default_model_ref
        self.providers_by_id = {provider.id: provider for provider in self.document.providers if provider.enabled}

    def resolve_model_ref(self, model_ref: str) -> ResolvedModelRef:
        provider_id, remote_model_id = model_ref.split("::", 1)
        provider = self.providers_by_id[provider_id]
        return ResolvedModelRef(provider=provider, remote_model_id=remote_model_id)

# backend/providers/gemini_native.py
import httpx


class GeminiNativeProvider:
    def __init__(self, client_factory=httpx.AsyncClient) -> None:
        self.client_factory = client_factory

    async def list_remote_models(self, *, api_key: str, base_url: str) -> list[dict]:
        async with self.client_factory() as client:
            response = await client.get(
                f"{base_url.rstrip('/')}/models",
                headers={"x-goog-api-key": api_key},
            )
        return [
            {
                "remoteId": item["name"].removeprefix("models/"),
                "label": item.get("displayName") or item["name"],
            }
            for item in response.json().get("models", [])
            if "generateContent" in item.get("supportedGenerationMethods", [])
        ]

    def build_tool_schema(self, tools: list[dict]) -> dict:
        return {
            "functionDeclarations": [
                {
                    "name": tool["name"],
                    "description": tool["description"],
                    "parameters": tool["parameters"],
                }
                for tool in tools
            ]
        }

    def normalize_tool_call_response(self, payload: dict) -> list[dict]:
        parts = payload["candidates"][0]["content"]["parts"]
        return [
            {
                "toolName": part["functionCall"]["name"],
                "callId": part["functionCall"]["name"],
                "arguments": dict(part["functionCall"].get("args") or {}),
                "rawProviderPayload": part,
            }
            for part in parts
            if "functionCall" in part
        ]

# backend/agent/tool_loop.py
from dataclasses import dataclass
from typing import Any

from backend.agent.tool_calling import ToolRegistry, ToolResultSlimmer


@dataclass(frozen=True)
class ToolLoopResult:
    content: str
    messages: list[dict[str, Any]]
    tool_rounds: int
    proposals: list[dict[str, Any]]


class ToolLoopOrchestrator:
    def __init__(self, *, registry: ToolRegistry, max_rounds: int = 4, slimmer: ToolResultSlimmer | None = None) -> None:
        self.registry = registry
        self.max_rounds = max_rounds
        self.slimmer = slimmer or ToolResultSlimmer()

    async def run(self, *, session, provider, messages: list[dict[str, Any]], model: str, **kwargs) -> ToolLoopResult:
        active_messages = list(messages)
        proposals: list[dict[str, Any]] = []
        for round_index in range(self.max_rounds + 1):
            response = await provider.generate_chat(messages=active_messages, model=model, tools=self.registry, **kwargs)
            tool_calls = response.get("toolCalls") or []
            if not tool_calls:
                return ToolLoopResult(content=response.get("text", ""), messages=active_messages, tool_rounds=round_index, proposals=proposals)

            normalized_calls = provider.normalize_tool_call_response(response.get("raw") or response)
            for tool_call in normalized_calls:
                tool_result = await self.registry.execute(session, tool_call["toolName"], tool_call["arguments"])
                if tool_call["toolName"].startswith("propose_") and isinstance(tool_result, dict) and isinstance(tool_result.get("proposal"), dict):
                    proposals.append(tool_result["proposal"])
                active_messages = provider.build_followup_messages_after_tool_result(
                    active_messages,
                    tool_call,
                    self.slimmer.slim(tool_call["toolName"], tool_result),
                )
        return ToolLoopResult(content="工具调用次数过多，请稍后重试或缩小问题范围。", messages=active_messages, tool_rounds=self.max_rounds, proposals=proposals)
